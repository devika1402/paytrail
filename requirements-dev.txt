ruff

sqlfluff  # SQL lint for the dbt models (jinja templater, no warehouse needed)

# Pin: dbt-core needs pathspec<0.13; mypy<2 is compatible with it. Holds the resolver
# to a version both accept (mypy 2.x would force >=1.0 and break dbt).
pathspec==0.12.1
